ИНКРЕМЕНТ И ДЕКРЕМЕНТ

var++ (постинкремент):

var a = 0, b = 10;
var a = b++;

ИТОГ 

a = 10
b = 11
___________________________________________________________________
++var (преикремент):

var a = 0, b = 10;
var a = ++b;

ИТОГ 

a = 11
b = 11
___________________________________________________________________
var-- (постдекремент):

var a = 0, b = 10;
var a = b--;

ИТОГ 

a = 10
b =  9
___________________________________________________________________
--var (предекремент):

var a = 0, b = 10;
var a = --b;

ИТОГ

a = 9
b = 9
